 1  #!/usr/bin/env python 
 2  """ 
 3  Convert ms output to a frequency spectrum. 
 4   
 5  Reads the ms results from stdin and outputs to stdout. 
 6   
 7  If the '-av' flag is passed, the output is the average spectrum of the ms 
 8  iterations. Otherwise it is the sum. 
 9   
10  For example: 'ms <details> | ms_jsfs.py -av > output.fs' 
11  """ 
12   
13  if __name__ == '__main__': 
14      import sys 
15      import dadi 
16   
17      average = ('-av' in sys.argv) 
18       
19      input = sys.stdin 
20      output = sys.stdout 
21      sfs,header = dadi.Spectrum.from_ms_file(input, average, mask_corners=True, 
22                                              return_header=True) 
23       
24      sfs.to_file(output, comment_lines=header)
